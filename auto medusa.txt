local Players = game:GetService("Players")

-- Settings
local MEDUSA_RANGE = 9.5          -- Attack radius (studs)
local MEDUSA_COOLDOWN = 0.12      -- Seconds between attacks

-- Internal variables
local autoMedusaEnabled = false
local autoMedusaConn = nil
local medusaCircle = nil
local medusaAttacking = false

-- Find Medusa's Head tool in character or backpack
local function findMedusaTool()
    local char = player.Character
    local backpack = player:FindFirstChildOfClass("Backpack")
    local keywords = {"medusa", "Medusa's Head", "medusa's head"}
    local function search(container)
        if not container then return nil end
        for _, tool in ipairs(container:GetChildren()) do
            if tool:IsA("Tool") then
                local nameLower = tool.Name:lower()
                for _, kw in ipairs(keywords) do
                    if nameLower:find(kw:lower()) then
                        return tool
                    end
                end
            end
        end
        return nil
    end
    return search(char) or search(backpack)
end

-- Unequip Medusa if it's currently held
local function unequipMedusa()
    local char = player.Character
    local humanoid = char and char:FindFirstChildOfClass("Humanoid")
    if not humanoid then return end
    local medusa = findMedusaTool()
    if medusa and medusa.Parent == char then
        pcall(function() humanoid:UnequipTools() end)
    end
end

-- Check if any enemy player is within range
local function enemyInRange()
    local char = player.Character
    local root = char and char:FindFirstChild("HumanoidRootPart")
    if not root then return false end
    for _, other in ipairs(Players:GetPlayers()) do
        if other ~= player and other.Character then
            local otherRoot = other.Character:FindFirstChild("HumanoidRootPart")
            local otherHum = other.Character:FindFirstChildOfClass("Humanoid")
            if otherRoot and otherHum and otherHum.Health > 0 then
                local distance = (root.Position - otherRoot.Position).Magnitude
                if distance <= MEDUSA_RANGE then
                    return true
                end
            end
        end
    end
    return false
end

-- Create the visual circle under the player
local function createMedusaCircle()
    if medusaCircle then medusaCircle:Destroy() end
    medusaCircle = Instance.new("Part")
    medusaCircle.Name = "AutoMedusaRange"
    medusaCircle.Shape = Enum.PartType.Cylinder
    medusaCircle.Size = Vector3.new(0.2, MEDUSA_RANGE * 2, MEDUSA_RANGE * 2)
    medusaCircle.Color = Color3.fromRGB(255, 0, 0)
    medusaCircle.Material = Enum.Material.Neon
    medusaCircle.Transparency = 0.35
    medusaCircle.Anchored = true
    medusaCircle.CanCollide = false
    medusaCircle.Parent = workspace
end

-- Update circle position every frame
local function updateMedusaCircle()
    local char = player.Character
    local root = char and char:FindFirstChild("HumanoidRootPart")
    if medusaCircle and root then
        medusaCircle.CFrame = CFrame.new(root.Position - Vector3.new(0, 2.6, 0)) * CFrame.Angles(0, 0, math.rad(90))
    end
end

-- Main auto‑medusa loop
local function startAutoMedusa()
    if autoMedusaConn then autoMedusaConn:Disconnect() end
    createMedusaCircle()
    autoMedusaConn = RunService.Heartbeat:Connect(function()
        if not autoMedusaEnabled then return end
        updateMedusaCircle()

        if not enemyInRange() then
            unequipMedusa()
            return
        end

        local medusa = findMedusaTool()
        if not medusa then return end

        local char = player.Character
        local humanoid = char and char:FindFirstChildOfClass("Humanoid")
        if humanoid and medusa.Parent ~= char then
            pcall(function() humanoid:EquipTool(medusa) end)
        end

        local heldTool = char and char:FindFirstChildOfClass("Tool")
        if not heldTool or medusaAttacking then return end
        medusaAttacking = true
        heldTool:Activate()
        task.delay(MEDUSA_COOLDOWN, function() medusaAttacking = false end)
    end)
end

local function stopAutoMedusa()
    if autoMedusaConn then autoMedusaConn:Disconnect(); autoMedusaConn = nil end
    if medusaCircle then medusaCircle:Destroy(); medusaCircle = nil end
    medusaAttacking = false
    unequipMedusa()
end

    statusLabel.Text = enabled and "ON" or "OFF"

    frame.BackgroundColor3 = enabled and Color3.fromRGB(8, 22, 12) or Color3.fromRGB(10, 5, 18)

    else
        stopAutoMedusa()
    end
